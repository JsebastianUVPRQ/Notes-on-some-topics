### 1.3. Función de descarga con control de tasa

```python
import requests
import time
from datetime import datetime, timedelta
import logging

logging.basicConfig(level=logging.INFO)

def fetch_premier_league(season_start_year: int) -> pd.DataFrame:
    """
    Descarga todos los partidos de la temporada especificada.
    season_start_year: año de inicio de la temporada (ej. 2025 para 2025/2026)
    """
    base_url = "https://api.football-data.org/v4"
    headers = {"X-Auth-Token": API_KEY}
    
    # Fechas: desde 1 de agosto hasta 31 de mayo del año siguiente
    date_from = f"{season_start_year}-08-01"
    date_to = f"{season_start_year+1}-05-31"
    
    endpoint = f"{base_url}/competitions/{PREMIER_LEAGUE_ID}/matches"
    params = {
        "dateFrom": date_from,
        "dateTo": date_to,
        "status": "FINISHED",
        "limit": 500  # máximo 500 por petición
    }
    
    all_matches = []
    while True:
        response = requests.get(endpoint, headers=headers, params=params)
        if response.status_code == 429:
            wait = int(response.headers.get("X-RequestCounter-Reset", 60))
            logging.warning(f"Rate limit alcanzado, esperando {wait}s")
            time.sleep(wait + 1)
            continue
        
        data = response.json()
        matches = data.get("matches", [])
        if not matches:
            break
        
        all_matches.extend(matches)
        
        # Paginación: respuesta incluye links para siguiente página?
        # La API v4 no usa paginación basada en páginas, sino que devuelve todos los partidos dentro del rango.
        # Si hay muchos, se pueden hacer varias llamadas filtrando por mes.
        break  # Salimos después de la primera petición
    
    # Convertir a DataFrame limpio
    records = []
    for m in all_matches:
        home = m["homeTeam"]["name"]
        away = m["awayTeam"]["name"]
        ft = m["score"]["fullTime"]
        if ft["home"] is None or ft["away"] is None:
            continue  # Descarta partidos no terminados
        
        home_goals = ft["home"]
        away_goals = ft["away"]
        
        records.append({
            "date": m["utcDate"][:10],
            "home_team": home,
            "away_team": away,
            "home_goals": home_goals,
            "away_goals": away_goals
        })
    
    return pd.DataFrame(records)
```

### 2.3. Definición del pipeline de backtesting parametrizable

```python
def run_backtest_step(context_size, n_estimators_ensemble, calibrator_alpha, 
                      features_to_use, bankroll=1000, kelly_fraction=0.25):
    """
    Ejecuta backtesting cronológico con los parámetros dados.
    Retorna métricas y resultados.
    """
    # Seleccionar features
    X = features_df[features_to_use].values
    y = outcomes  # 0: local, 1: empate, 2: visitante
    
    # Orden cronológico (ya está)
    dates = pd.to_datetime(features_df.index.get_level_values('date'))
    uniq_dates = sorted(dates.unique())
    
    # Calibrador y modelo
    model = TabPFNEnsemble(n_estimators=n_estimators_ensemble, device=DEVICE)
    calibrator = DirichletCalibrator(alpha_init=calibrator_alpha)
    
    # Backtest engine (necesita probabilidades y cuotas)
    backtester = BacktestEngine(model, calibrator,
                                portfolio_manager=RobustPortfolioManager(
                                    initial_bankroll=bankroll,
                                    kelly_fraction=kelly_fraction,
                                    entropy_threshold=0.3
                                ),
                                odds=odds,
                                use_calibrator=True,
                                context_size=context_size)
    
    # Recorrer fechas
    results = backtester.run(X, y, dates)
    
    # Métricas
    roi = results['roi']
    log_loss = results['log_loss']
    rps = results['rps']
    total_bets = results['total_bets']
    final_bankroll = results['final_bankroll']
    sharpe = results.get('sharpe', None)
    
    return {
        'roi': roi,
        'log_loss': log_loss,
        'rps': rps,
        'total_bets': total_bets,
        'final_bankroll': final_bankroll,
        'sharpe': sharpe,
        'pnl_series': results['pnl_series'],
        'calibration_curve': results['calibration_curve']
    }
```

### 2.4. Optimización con Optuna + MLflow

```python
def objective(trial):
    # Parámetros a optimizar
    context_size = trial.suggest_int('context_size', 300, 2000, step=100)
    n_estimators = trial.suggest_int('n_estimators', 1, 5)
    calibrator_alpha = trial.suggest_float('calibrator_alpha', 1.0, 5.0)
    
    # Selección de features vía importancia de permutación (una pasada rápida con Lasso)
    # Pero aquí lo simplificamos: usamos todas las features base por ahora.
    features_to_use = BASE_FEATURES  # lista de strings
    
    # Ejecutar backtest con validación cruzada temporal (últimas 2 temporadas como test)
    # Para ahorrar tiempo, hacemos un solo fold: train en 23/24, test en 24/25
    mask_train = features_df.index.get_level_values('date') < '2025-07-01'
    mask_test = features_df.index.get_level_values('date') >= '2025-07-01'
    
    X_train = features_df[mask_train][features_to_use].values
    y_train = outcomes[mask_train]
    X_test = features_df[mask_test][features_to_use].values
    y_test = outcomes[mask_test]
    odds_test = odds[mask_test]
    
    if len(X_test) == 0:
        return 0.0
    
    model = TabPFNEnsemble(n_estimators=n_estimators, device=DEVICE)
    calibrator = DirichletCalibrator(alpha_init=calibrator_alpha)
    
    backtester = BacktestEngine(model, calibrator,
                                portfolio_manager=None,  # Sin apuestas, solo probabilidades
                                odds=odds_test,
                                use_calibrator=True,
                                context_size=context_size)
    # Simular predicciones
    probs = backtester.predict_all(X_train, y_train, X_test)
    
    # Log Loss
    from sklearn.metrics import log_loss
    ll = log_loss(y_test, probs, labels=[0,1,2])
    return ll

# Configurar callback de MLflow
mlflow_callback = MLflowCallback(
    tracking_uri=mlflow.get_tracking_uri(),
    experiment_name="betedge_premier_league"
)

study = optuna.create_study(direction='minimize')
study.optimize(objective, n_trials=50, callbacks=[mlflow_callback])

print("Mejores parámetros:", study.best_params)
print("Mejor Log Loss:", study.best_value)
```
